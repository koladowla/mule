# Build the image
docker build -t mule-job-mail .

# Stop and remove existing container (if running)
docker stop mule-job-mail && docker rm mule-job-mail

# Run with persistent logs
docker run -d --name mule-job-mail --restart unless-stopped -v %CD%\logs:/opt/mule/logs mule-job-mail

# Monitor logs
docker logs -f mule-job-mail

# Verify container status
docker ps

# Check timezone inside container
docker exec -it mule-job-mail date
